  commentText: any;

  feedData: Feed[] = [{

  addComment():void{
    if (this.commentText) {
      const commentObj: Feed = {
        'authorName': 'User5',
        'authorLink': '',
        'action': 'posted on page',
        'message': this.commentText,
        'attachments': [],
        'likes': 0
      }
      this.feedData.push(commentObj);
      this.commentText ='';
    }
    
  }

}
